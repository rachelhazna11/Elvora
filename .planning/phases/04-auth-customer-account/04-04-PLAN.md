      setTab(tab) { this.activeTab = tab; location.hash = tab; },

      async init() {
        const { data: { user } } = await window.supabase.auth.getUser();
        if (!user) { window.location.href = '/auth.html?tab=signin'; return; }
        this._currentUser = user;
        const hash = location.hash.replace('#', '');
        if (['profile', 'wishlist', 'preferences'].includes(hash)) this.activeTab = hash;
        this.loading = false;
        await Promise.all([ this.loadProfile(user), this.loadWishlist(user) ]);
      },

      async loadProfile(user) {
        const { data } = await window.supabase
          .from('user_profiles')
          .select('first_name, last_name, preferred_activity, fit_preference, style_aesthetic, colour_preference')
          .eq('id', user.id)   // CORRECT: 'id' not 'user_id' — schema correction applied
          .single();
        if (data) {
          this.profile = { first_name: data.first_name || '', last_name: data.last_name || '', email: user.email };
          this.prefs = { activity: data.preferred_activity, fit: data.fit_preference, aesthetic: data.style_aesthetic, colour: data.colour_preference };
        } else {
          this.profile = { first_name: user.user_metadata?.first_name || '', last_name: user.user_metadata?.last_name || '', email: user.email };
        }
      },

      async saveProfile() {
        this.profileSaving = true;
        const { error } = await window.supabase
          .from('user_profiles')
          .update({ first_name: this.profile.first_name, last_name: this.profile.last_name })
          .eq('id', this._currentUser.id);  // CORRECT: 'id' not 'user_id'
        if (error) {
          Toastify({ text: "Couldn't save — try again", duration: 3000, gravity: 'bottom', position: 'right',
            style: { background: 'var(--rose)', color: '#fff', borderRadius: '50px', fontSize: '12px', letterSpacing: '2px' } }).showToast();
        } else {
          Toastify({ text: 'Changes saved', duration: 2000, gravity: 'bottom', position: 'right',
            style: { background: 'var(--sage)', color: '#fff', borderRadius: '50px', fontSize: '12px', letterSpacing: '2px' } }).showToast();
        }
        this.profileSaving = false;
      },

      async loadWishlist(user) {
        this.wishlistLoading = true;
        const { data } = await window.supabase
          .from('wishlist_items')
          .select('id, product_id, products(id, name, slug, base_price, product_images(url, alt_text, display_order))')
          .eq('user_id', user.id);   // CORRECT: wishlist_items uses 'user_id' (only user_profiles uses 'id')
        this.wishlistItems = (data || []).map(row => {
          const images = (row.products?.product_images || []).sort((a, b) => a.display_order - b.display_order);
          return { wishlistId: row.id, id: row.products?.id, name: row.products?.name, slug: row.products?.slug, base_price: row.products?.base_price, image: images[0]?.url || '' };
        });
        this.wishlistLoading = false;
      },

      async removeFromWishlist(wishlistId, productId) {
        // Optimistic remove
        const prev = [...this.wishlistItems];
        this.wishlistItems = this.wishlistItems.filter(item => item.wishlistId !== wishlistId);
        const { error } = await window.supabase
          .from('wishlist_items')
          .delete()
          .eq('id', wishlistId);
        if (error) {
          this.wishlistItems = prev;  // revert
          Toastify({ text: "Couldn't remove item — try again", duration: 3000, gravity: 'bottom', position: 'right',
            style: { background: 'var(--rose)', color: '#fff', borderRadius: '50px', fontSize: '12px', letterSpacing: '2px' } }).showToast();
        }
      },

      async savePrefs() {
        this.prefsSaving = true;
        const { error } = await window.supabase
          .from('user_profiles')
          .update({ preferred_activity: this.prefs.activity, fit_preference: this.prefs.fit, style_aesthetic: this.prefs.aesthetic, colour_preference: this.prefs.colour })
          .eq('id', this._currentUser.id);  // CORRECT: 'id' not 'user_id'
        if (error) {
          Toastify({ text: "Couldn't save — try again", duration: 3000, gravity: 'bottom', position: 'right',
            style: { background: 'var(--rose)', color: '#fff', borderRadius: '50px', fontSize: '12px', letterSpacing: '2px' } }).showToast();
        } else {
          Toastify({ text: 'Preferences saved', duration: 2000, gravity: 'bottom', position: 'right',
            style: { background: 'var(--sage)', color: '#fff', borderRadius: '50px', fontSize: '12px', letterSpacing: '2px' } }).showToast();
        }
        this.prefsSaving = false;
      }
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → Supabase DB (user_profiles) | Reads/writes profile and preferences for the authenticated user |
| Browser → Supabase DB (wishlist_items) | Reads/deletes wishlist rows |
| Alpine x-data → DOM | User-provided first/last name rendered via x-text (escaped) |
| page load → getUser() | Validates session with Supabase before rendering data |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-04-01 | Elevation of Privilege | account.html unauthenticated access | mitigate | init() calls getUser() → redirects to auth.html if null; RLS backs this up — even if redirect is bypassed, all queries return empty (auth.uid() = id policy on user_profiles) |
| T-04-04-02 | Elevation of Privilege | Cross-user profile access | mitigate | All user_profiles queries use .eq('id', user.id) + RLS policy auth.uid() = id — server enforced |
| T-04-04-03 | Elevation of Privilege | Cross-user wishlist access | mitigate | wishlist_items RLS: auth.uid() = user_id enforced at DB; client .eq('user_id', user.id) is defense-in-depth |
| T-04-04-04 | Tampering | x-text for profile name display | mitigate | Alpine x-text escapes HTML — profile.first_name rendered as text, not HTML |
| T-04-04-05 | Tampering | Style preference values | accept | Values are from a closed set of chip options; free-text input not accepted; server does not enforce enum constraint but values are UI-controlled |
| T-04-04-06 | Denial of Service | Wishlist loaded on every account.html load | accept | Single JOIN query; no pagination needed at MVP scale |
| T-04-04-SC | Tampering | npm/pip/cargo installs | accept | No new packages; SQL migration is DDL only |
</threat_model>
