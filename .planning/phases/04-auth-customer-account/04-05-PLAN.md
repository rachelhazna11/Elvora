    async initWishlist() {
      const { data: { user } } = await window.supabase.auth.getUser();
      if (!user) return;
      const { data } = await window.supabase
        .from('wishlist_items')
        .select('product_id')

    }

    New method — toggleWishlist(productId):
    async toggleWishlist(productId) {
      const { data: { user } } = await window.supabase.auth.getUser();
      if (!user) {
        Toastify({
          text: 'Sign in to save items to your wishlist',
          duration: 3000,
          gravity: 'bottom', position: 'right',
          style: { background: 'var(--charcoal)', color: '#fff', borderRadius: '50px', fontSize: '12px', letterSpacing: '2px' },
          onClick() { window.location.href = '/auth.html?tab=signin'; }
        }).showToast();
        setTimeout(() => { window.location.href = '/auth.html?tab=signin'; }, 2000);
        return;
      }
      const wasWishlisted = this.wishlistedIds.has(productId);
      if (wasWishlisted) {
        this.wishlistedIds = new Set([...this.wishlistedIds].filter(id => id !== productId));
        const { error } = await window.supabase.from('wishlist_items').delete().eq('user_id', user.id).eq('product_id', productId);
        if (error) {
          this.wishlistedIds = new Set([...this.wishlistedIds, productId]);
          Toastify({ text: "Couldn't save item — try again", duration: 3000, gravity: 'bottom', position: 'right',
            style: { background: 'var(--rose)', color: '#fff', borderRadius: '50px', fontSize: '12px', letterSpacing: '2px' } }).showToast();
        }
      } else {
        this.wishlistedIds = new Set([...this.wishlistedIds, productId]);
        const { error } = await window.supabase.from('wishlist_items').insert({ user_id: user.id, product_id: productId });
        if (error) {
          this.wishlistedIds = new Set([...this.wishlistedIds].filter(id => id !== productId));
          Toastify({ text: "Couldn't save item — try again", duration: 3000, gravity: 'bottom', position: 'right',
            style: { background: 'var(--rose)', color: '#fff', borderRadius: '50px', fontSize: '12px', letterSpacing: '2px' } }).showToast();
        }
      }
    }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| PLP/PDP heart click → Supabase wishlist_items | user_id and product_id cross this boundary on insert/delete |
| Guest click → auth.html redirect | guest session state used for redirect decision |
| Alpine Set state → DOM | wishlistedIds.has() drives visual state; no user data rendered via x-text |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-05-01 | Elevation of Privilege | Cross-user wishlist insert | mitigate | INSERT sends { user_id: user.id } AND Supabase RLS INSERT policy enforces auth.uid() = user_id — server blocks cross-user inserts |
| T-04-05-02 | Elevation of Privilege | Cross-user wishlist delete | mitigate | DELETE uses .eq('user_id', user.id).eq('product_id', productId) AND RLS enforces auth.uid() = user_id — cannot delete another user's wishlist item |
| T-04-05-03 | Tampering | Optimistic UI revert failure | accept | If Supabase returns error but network is unreliable, Set is reverted to correct state; worst case is a briefly stale heart icon that refreshes on next load |
| T-04-05-04 | Spoofing | Guest redirect to auth page | accept | Redirect is UI-only; no security operation performed; guest clicking heart cannot access wishlist data (RLS blocks all wishlist_items queries without valid session) |
| T-04-05-05 | Denial of Service | initWishlist() query on every PLP/PDP load | accept | Single SELECT on wishlist_items with user_id filter; fast, indexed; acceptable at MVP scale |
| T-04-05-SC | Tampering | npm/pip/cargo installs | accept | No new packages; no installs in this plan |
</threat_model>
