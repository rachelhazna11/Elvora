<tasks>

<task type="auto">
  <name>Task 1: Create AI Sessions Table Migration</name>
  <files>supabase/migrations/004_style_sessions.sql</files>
  <action>
    Create the `ai_style_sessions` table with the following columns:
    - `id`: uuid (primary key)
    - `user_id`: uuid (foreign key to auth.users, nullable for guests)
    - `preferences`: jsonb (activity, fit, aesthetic, colour)
    - `recommendations`: jsonb (list of outfits with product_ids and guidance)
    - `colour_guidance`: text
    - `created_at`: timestamp with time zone (default now())
    Enable RLS:
    - Authenticated users can insert their own sessions.
    - Authenticated users can select their own sessions.
    - (Optional) Service role can clean up old sessions.
  </action>
  <verify>
    <automated>test -f supabase/migrations/004_style_sessions.sql</automated>
  </verify>
  <done>Database schema supports session persistence.</done>
</task>

<task type="auto">
  <name>Task 2: Implement Session Saving in Edge Function</name>
  <files>supabase/functions/style-match/index.ts</files>
  <action>
    Update the `style-match` Edge Function to save the AI results to the `ai_style_sessions` table if a user is authenticated.
    - Extract `user_id` from the JWT (Authorization header).
    - If `user_id` is present, insert a row into `ai_style_sessions` before returning the response.
  </action>
  <verify>
    <automated>grep -c "from('ai_style_sessions').insert" supabase/functions/style-match/index.ts</automated>
  </verify>
  <done>AI sessions are persisted in the database for authenticated users.</done>
</task>

<task type="auto">
