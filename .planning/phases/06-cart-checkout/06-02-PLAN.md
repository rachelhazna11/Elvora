# Plan 06-02: cart.js + auth.js Extensions — Drawer State and Supabase Sync

## Goal
`js/cart.js` exposes `cartDrawerOpen`, `openDrawer()`, `closeDrawer()`, `mergeGuestCartToSupabase()`, `loadCartFromSupabase()`, and `loadFromStorage()` as named exports; `js/auth.js` passes the event name as a second argument to `onAuthChange` callbacks; and `js/components.js` wires auth-change events to trigger cart merge or load from Supabase — enabling the full cart persistence cycle that Plans 06-03 through 06-05 depend on.

## Context
`js/cart.js` currently registers `Alpine.store('cart')` via `initCartStore()` called on `alpine:init`. The store has `add`, `remove`, `setQty`, `clear` methods and `count`/`total`/`totalFormatted` getters. `loadFromStorage()` and `saveToStorage()` are private functions (not exported). The `initCartStore()` function is not exported either.

`js/auth.js` line 71: `onAuthChange(callback)` subscribes to `supabase.auth.onAuthStateChange((_event, session) => callbacks.forEach(cb => cb(session?.user ?? null)))`. The `_event` name is discarded — it must be passed through so `components.js` can distinguish `SIGNED_IN` from `INITIAL_SESSION`.

`js/components.js` line 26: `onAuthChange((user) => { Alpine.store('auth').user = user; Alpine.store('auth').loggedIn = !!user; })` — must be extended to accept `(user, event)` and trigger merge or load accordingly.

`window.supabase` is already exposed globally in `components.js` line 9 (`window.supabase = supabase`), so Supabase calls in cart sync functions can use either the imported `supabase` from `./supabase.js` or `window.supabase`.

**1c. Add `mergeGuestCartToSupabase` as a named exported async function** — append after `initCartStore()`:

```js
/**
 * Called on SIGNED_IN: merges localStorage cart items into Supabase cart_items,
 * then loads the authoritative cart from Supabase into Alpine store.
 */
export async function mergeGuestCartToSupabase(user) {
  const { supabase } = await import('./supabase.js');
  const localItems = loadFromStorage();
  if (!localItems.length) {
    await loadCartFromSupabase(user.id);
    return;
  }

  // Build upsert payload — variant_id may be null for products without variants
  const upsertRows = localItems.map(item => ({
    user_id: user.id,
    product_id: item.productId,
    variant_id: item.variantId || null,
    quantity: item.qty,
  }));

  // Supabase upsert with onConflict on (user_id, product_id, variant_id).
  // This SUMS quantities for existing rows via a DB trigger or we handle it manually.
  // Strategy: fetch existing rows first, then sum quantities, then upsert.
  const { data: existing } = await supabase
    .from('cart_items')
    .select('id, product_id, variant_id, quantity')
    .eq('user_id', user.id);

  const existingMap = new Map(
    (existing || []).map(row => [`${row.product_id}|${row.variant_id || ''}`, row])
  );

  const merged = upsertRows.map(row => {
    const key = `${row.product_id}|${row.variant_id || ''}`;
    const existingRow = existingMap.get(key);
    return existingRow
      ? { ...row, id: existingRow.id, quantity: existingRow.quantity + row.quantity }
      : row;
  });

  await supabase
    .from('cart_items')
    .upsert(merged, { onConflict: 'user_id,variant_id' });

  await loadCartFromSupabase(user.id);
}
```

**1d. Add `loadCartFromSupabase` as a named exported async function** — append after `mergeGuestCartToSupabase`:

```js
/**
 * Called on INITIAL_SESSION: loads cart from Supabase and overwrites Alpine store + localStorage.
 * Joins cart_items -> product_variants -> products -> product_images.
 */
export async function loadCartFromSupabase(userId) {
  const { supabase } = await import('./supabase.js');

  const { data: rows, error } = await supabase
    .from('cart_items')
    .select(`
      id,
      quantity,
      product_id,
      variant_id,
      product_variants ( id, colour, size ),
      products (
        id, name, slug, base_price,
        product_images ( url, display_order )
      )
    `)
    .eq('user_id', userId);

  if (error) {
    console.warn('[cart] loadCartFromSupabase error:', error.message);
    return;
  }

  const items = (rows || []).map(row => {
    const variant = row.product_variants;
    const product = row.products;
    const images = [...(product?.product_images || [])].sort((a, b) => a.display_order - b.display_order);
    const image = images[0]?.url || '';
    const colour = variant?.colour || '';
    const size = variant?.size || '';
    const key = `${product?.id}|${row.variant_id || ''}|${size}`;
    return {
      key,
      productId: product?.id,
      variantId: row.variant_id,
      name: product?.name || '',
      slug: product?.slug || '',
      price: product?.base_price || 0,
      image,
      colour,
      size,
      qty: row.quantity,
    };
  });

  // Replace store and localStorage
  if (window.Alpine && Alpine.store('cart')) {
    Alpine.store('cart').items = items;
  }
  localStorage.setItem('elvora_cart', JSON.stringify(items));
}
```

### Task 2: Extend `js/auth.js` — pass event name to callbacks

**File:** `js/auth.js`
**Action:** modify

Change the `onAuthChange` function (lines 70–75) to pass the event name as the second argument to each callback. Change:

```js
// BEFORE:
export function onAuthChange(callback) {
  const { data: { subscription } } = supabase.auth.onAuthStateChange((_event, session) => {
    callback(session?.user ?? null);
  });
  return subscription;
}

// AFTER:
export function onAuthChange(callback) {
  const { data: { subscription } } = supabase.auth.onAuthStateChange((event, session) => {
    callback(session?.user ?? null, event);
  });
  return subscription;
}
```

This is a non-breaking change: existing callers that only use the first argument (user) continue to work. Only callers that need the event need to add a second parameter.

```js
// BEFORE:
async function initAuth() {
  const user = await getUser();
  Alpine.store('auth').user = user;
  Alpine.store('auth').loggedIn = !!user;
  onAuthChange((user) => {
    Alpine.store('auth').user = user;
    Alpine.store('auth').loggedIn = !!user;
  });
}

// AFTER:
async function initAuth() {
  const user = await getUser();
  Alpine.store('auth').user = user;
  Alpine.store('auth').loggedIn = !!user;

  // If already logged in on page load, load cart from Supabase
  if (user) {
    loadCartFromSupabase(user.id).catch(err => console.warn('[cart] initial load failed:', err));
  }

  onAuthChange(async (user, event) => {
    Alpine.store('auth').user = user;
    Alpine.store('auth').loggedIn = !!user;

    if (user && event === 'SIGNED_IN') {
      // User just signed in — merge guest localStorage cart into Supabase
      try { await mergeGuestCartToSupabase(user); } catch (err) { console.warn('[cart] merge failed:', err); }
    } else if (user && event === 'INITIAL_SESSION') {
      // Tab reload with active session — load Supabase cart
      try { await loadCartFromSupabase(user.id); } catch (err) { console.warn('[cart] load failed:', err); }
    } else if (!user) {
      // Signed out — restore from localStorage (may be empty)
      const localItems = loadFromStorage();
      if (window.Alpine && Alpine.store('cart')) {
        Alpine.store('cart').items = localItems;
      }
    }
  });
}
```
